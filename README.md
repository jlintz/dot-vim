# Setup
 - Clone repo
 - `ln -s ~/dot-vim ~/.vim`
 - `ln -s ~/dot-vim/.vimrc ~/.vimrc`
 - `cd ~/.vim`
 - `brew install ag`
 - `brew install nodejs`
 - `pip install jedi`
 - `pip install flake8`

# Installing plugins
:PlugInstall

# Updating plugins
:PlugUpdate

# Powerline fonts
Install https://github.com/powerline/fonts/blob/master/Meslo%20Slashed/Meslo%20LG%20M%20Regular%20for%20Powerline.ttf
